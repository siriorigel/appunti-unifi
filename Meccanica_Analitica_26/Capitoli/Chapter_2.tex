\section{Curve in $\mathbb{R}^3$}

\begin{definition}[Curva]
    Una \textbf{curva} in $\mathbb{R}^3$ è definita come un'appliciazione continua da un intervallo aperto $U \subset \mathbb{R}$ nella sua immagine $\mathbb{R}^3$
    ovvero, utilizzando le notazioni introdotte nella Sezione \ref{sec:1_vettori}:
\end{definition}

La coppia $(t, U)$ dove $t$ è la variabile $t\in U$ utilizzata per percorrere la curva, è detta \textbf{parametrizzazione} della curva. 
La variabile $t$ è detta \textbf{ascissa curvilinea} o parametro d'arco, mentre $U$ è detto \textbf{dominio} della curva. L'insieme $\vec r(U) \subset \mathbb{R}^3$ è detto \textbf{immagine} della curva.

\begin{example}[Parabola]
    La parabola è una curva che può essere rappresentata da una parametrizzazione del tipo:
    \begin{equation*}
        \vec r(t) = t\hat i + t^2 \hat j + 0\hat k
    \end{equation*}
    con $t \in \mathbb{R}$. In questo caso, il dominio $U$ è l'intero insieme dei numeri reali, e l'immagine della curva è l'insieme dei punti $(t, t^2, 0)$ che formano una parabola nel piano $xy$.

    \begin{figure}[H]
    \centering
    \begin{tikzpicture}[scale=1]
    
    % Assi
    \draw[->, thick] (-3,0) -- (3,0) node[right] {$x$};
    \draw[->, thick] (0,-0.5) -- (0,4) node[above] {$y$};
    
    % Parabola
    \draw[blue, thick, domain=-2:2, smooth] plot (\x, {\x*\x});
    \node[blue] at (2.5,3) {$y = x^2$};
    
    % Punto di esempio
    \filldraw[red] (1.5,2.25) circle (2pt);
    \node[red, above right] at (1.5,2.25) {$(t, t^2)$};
    
    \end{tikzpicture}
    \end{figure}
    
\end{example}

\begin{example}[Semicirconferenza]
    Una semicirconferenza nel piano $xy$ può essere rappresentata da:
    \begin{equation*}
        \vec r(t) = \cos(t)\hat i + \sin(t)\hat j + 0\hat k
    \end{equation*}
    con $t \in [0, \pi]$. In questo caso, il dominio $U$ è l'intervallo chiuso $[0, \pi]$, e l'immagine della curva è l'insieme dei punti $(\cos(t), \sin(t), 0)$ che formano una semicirconferenza nel piano $xy$.

    \begin{figure}[H]
    \centering
    \begin{tikzpicture}[scale=1]
    
    % Assi
    \draw[->, thick] (-1.5,0) -- (1.5,0) node[right] {$x$};
    \draw[->, thick] (0,-0.5) -- (0,1.5) node[above] {$y$};
    
    % Semicirconferenza
    \draw[blue, thick, domain=0:pi, smooth] plot ({cos(\x r)}, {sin(\x r)});
    
    % Punto iniziale e finale
    \filldraw[red] (1,0) circle (2pt);
    \filldraw[red] (0,1) circle (2pt);
    
    % Etichette
    \node[red] at (1.2,-0.2) {$(1, 0)$};
    \node[red] at (-0.2,1.2) {$(0, 1)$};
    
    \end{tikzpicture}
    \end{figure}
    
    Si può anche rappresentare la stessa semicirconferenza con una parametrizzazione diversa, ad esempio:
    \begin{equation*}
        \vec r(t) = \hat i t + \hat j \sqrt{1 - t^2}
    \end{equation*}
    con $t \in [-1, 1]$. In questo caso, il dominio $U$ è l'intervallo chiuso $[-1, 1]$, e l'immagine della curva è la stessa semicirconferenza nel piano $xy$.
\end{example}

\begin{definition}
    Supponiamo di avere due parametri $t_1$ e $t_2$ che vivono su $U_1$ e $U_2$. Se esiste una funzione $t_1(t_2): U_2 \to U_1$ che è biunivoca e tale che $\vec r_1(t_1) = \vec r_2(t_2(t_1))$ per ogni $t_1 \in U_1$, allora diciamo che le curve $\vec r_1$ e $\vec r_2$ sono \textbf{equivalenti} o \textbf{parametrizzazioni diverse della stessa curva}.
\end{definition}
